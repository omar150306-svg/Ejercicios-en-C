#include <stdio.h>

int main()
{
    int nu=5;//nu representa al número secreto
    int op;//op es el número que ingresa el usuario
    do{
    printf("Adivina el número secreto:\n");
    scanf("%d", &op);
     if( op < nu){
         printf("El número secreto es mayor. Intenta de nuevo por favor.\n");
     }else if( op > nu){
         printf("El número secreto es menor. Intenta de nuevo por favor.\n");
     }else{
         printf("ERROR. Caracter ingresado inválido.\n");
     }
    
    }while( op != nu);{
        printf("Correcto, el número secreto era %d", nu);
    }
    

    return 0;
}
